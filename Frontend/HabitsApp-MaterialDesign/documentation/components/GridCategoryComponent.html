import { Component,ViewChild, OnInit, Output, EventEmitter } from '@angular/core';
import {MatTableDataSource, MatPaginator} from  '@angular/material';
import { HabitsWorkoutService } from '../habits-workout.service';
import * as _ from 'lodash';

@Component({
  selector: 'app-grid-category',
  templateUrl: './grid-category.component.html',
  styleUrls: ['./grid-category.component.css']
})
export class GridCategoryComponent implements OnInit {
  @Output() recordDeleted = new EventEmitter<any>();
  public habitsCategoryData1: Array<any>;
  public currentCategoryHabits: any;
  loading  =  true;
  displayedColumns = ["id", "category"]; 
  displayedColumns1 = ["name", "description","habitsdate","operations"]; 
  MyDataSource: any;
  MyDataSource_Sport: any;
  @ViewChild(MatPaginator) paginator:  MatPaginator;


  category =[
    {"id": 1, "name": "Sport"},
    {"id": 2, "name": "Health"},
    {"id": 3, "name": "Education"},
    {"id": 4, "name": "Work"},
    {"id": 5, "name": "Home"},
  ]

  constructor( private habitsworkoutService: HabitsWorkoutService) {
  
    habitsworkoutService.get().subscribe((data2: any) => {
      this.MyDataSource_Sport = new MatTableDataSource();  
      this.MyDataSource_Sport.data = data2;
      this.MyDataSource_Sport.paginator  =  this.paginator; 
     });
  }
  applyFilter(filterValue: string) {
    this.MyDataSource_Sport.filter = filterValue.trim().toLowerCase();
  }

  ngOnInit():  void {
    setTimeout(() => {this.loading  =  false;}, 1000);
 
  } 

  public deleteRecord(record) {
    this.recordDeleted.emit(record);
   const deleteIndex = _.findIndex(this.MyDataSource_Sport.data, {id: record.id});  
   this.habitsworkoutService.remove(record).subscribe(
    result => this.MyDataSource_Sport.data.splice(deleteIndex, 1)
   );
   location.reload();
  }
}
