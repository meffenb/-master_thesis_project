  public habitsData: any;
  public MyDataSource1: any;
  public habitsProgressData1: Array<any>;
  public currentProgressHabits: any;
  loading  =  true;
  displayedColumns = ["id", "data", "status","habitsID", "operations"]; 
  MyDataSource: any;
  @ViewChild(MatPaginator) paginator:  MatPaginator;

  data: any;

  cc: any;
  i:any;
size:any;
  constructor(private habitsprogressService: HabitsProgressService,private habitsworkoutService: HabitsWorkoutService,  public dialog: MatDialog) {
    habitsprogressService.getProgress().subscribe((data1: any) => {
      this.MyDataSource = new MatTableDataSource();  
      this.MyDataSource.data = data1;
      this.MyDataSource.paginator  =  this.paginator; 
      this.MyDataSource1 = data1;
this.size = this.MyDataSource1.length;
    });

    habitsworkoutService.get().subscribe((data: any) =>{ 
      this.habitsData = data;});


  }
  applyFilter(filterValue: string) {
    this.MyDataSource.filter = filterValue.trim().toLowerCase();
  }

  ngOnInit():  void {
    setTimeout(() => {this.loading  =  false;
    
this.cc=[0,0,0];
 for (this.i=0; this.i<this.size; this.i++){
if(this.MyDataSource1[this.i].habitsId == this.habitsData[0].id){
  this.cc[0]++
}
 };
 for (this.i=0; this.i<this.size; this.i++){
if(this.MyDataSource1[this.i].habitsId == this.habitsData[1].id){
  this.cc[1]++
}
 };
 for (this.i=0; this.i<this.size; this.i++){
if(this.MyDataSource1[this.i].habitsId == this.habitsData[2].id){
  this.cc[2]++
}
 };

      this.data = {
        labels: [this.habitsData[0].name,this.habitsData[1].name,this.habitsData[2].name],
        datasets: [
            {
                data: [this.cc[0], this.cc[1], this.cc[2]],
                backgroundColor: [
                    "#FF6384",
                    "#36A2EB",
                    "#FFCE56"
                ],
                hoverBackgroundColor: [
                    "#FF6384",
                    "#36A2EB",
                    "#FFCE56"
                ]
            }]    
        };
  
        this.data1 = {
          labels: [ 'March', 'April', 'May'],
          datasets: [
              {
                  label: this.habitsData[0].name,
                  backgroundColor: "#FF6384",
                  borderColor: "#FF6384",
                  data: [0, 2, 3],
                  fill: false
              },
              {
                label: this.habitsData[1].name,
                backgroundColor: "#36A2EB",
                borderColor: "#36A2EB",
                data: [0, 0, 1],
                fill: false
            },
              {
                  label: this.habitsData[2].name,
                  backgroundColor: "#FFCE56",
                  borderColor: "#FFCE56",
                  data: [0, 0, 1],
                  fill: false
              }
          ]
      }
    }, 1000);
 
  } 
  
    public deleteProgressRecord(progressrecord) {
     this.recordProgressDeleted.emit(progressrecord);
    const deleteIndex = _.findIndex(this.habitsProgressData1, {id: progressrecord.id});  
    this.habitsprogressService.removeProgress(progressrecord).subscribe(
     result => this.habitsProgressData1.splice(deleteIndex, 1)
    );
    location.reload();
    }
    
  
     public newProgressRecord() {
      const dialogRef = this.dialog.open(Dialogaddprogresshabits, {
        width: '600px'
      });
    }
    
  }


    @Component({
      selector: 'dialog-add-progress',
      templateUrl: './dialog-add-progress.html',
      styleUrls: ['./grid-progress.component.css']
    })
    export class Dialogaddprogresshabits {
      public habitsData: Array<any>;
      public currentHabits: any;
    
      constructor( public dialogRef: MatDialogRef<Dialogaddprogresshabits>,private habitsprogressService: HabitsProgressService) { }
    
      public createHabitsProgress = function(progresshabits: any) {
        this.habitsprogressService.addProgress(progresshabits).subscribe(
          habitsProgressRecord =>{progresshabits.id =habitsProgressRecord.id;
            this.habitsProgressData.pushProgress(progresshabits)})
            this.dialogRef.close();
                location.reload();
      }

      cancelClick(): void {
        this.dialogRef.close();
      }
    }
